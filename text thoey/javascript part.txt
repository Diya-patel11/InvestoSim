// Initialize user's virtual money (for the first time) if not set
if (!localStorage.getItem('availableCash')) {
    localStorage.setItem('availableCash', 5000); // Starting balance $5000
}

// Initialize total portfolio value (for the first time) if not set
if (!localStorage.getItem('portfolioValue')) {
    localStorage.setItem('portfolioValue', 0); // Initial portfolio value
}

// Function to update the displayed portfolio value and available cash
function updatePortfolio() {
    const availableCash = localStorage.getItem('availableCash');
    const portfolioValue = localStorage.getItem('portfolioValue');
    
    document.getElementById('availableCash').textContent = `$${availableCash}`;
    document.getElementById('portfolioValue').textContent = `$${portfolioValue}`;
}

// Example trade function (buy/sell stock)
function tradeStock(action) {
    let availableCash = parseFloat(localStorage.getItem('availableCash'));
    let portfolioValue = parseFloat(localStorage.getItem('portfolioValue'));
    let stockPrice = 100; // Example stock price (can be dynamic)

    if (action === 'buy') {
        // Buy stock logic (deduct money from available cash)
        if (availableCash >= stockPrice) {
            availableCash -= stockPrice;
            portfolioValue += stockPrice; // Increase portfolio value by stock value
            localStorage.setItem('availableCash', availableCash);
            localStorage.setItem('portfolioValue', portfolioValue);
            alert("Stock bought successfully!");
        } else {
            alert("Insufficient funds to buy stock.");
        }
    } else if (action === 'sell') {
        // Sell stock logic (add money to available cash)
        if (portfolioValue >= stockPrice) {
            availableCash += stockPrice;
            portfolioValue -= stockPrice; // Decrease portfolio value by stock value
            localStorage.setItem('availableCash', availableCash);
            localStorage.setItem('portfolioValue', portfolioValue);
            alert("Stock sold successfully!");
        } else {
            alert("You don't own enough stocks to sell.");
        }
    }

    // Update the displayed values
    updatePortfolio();
}

// Call the function to update portfolio when the page loads
window.onload = updatePortfolio;

const API_BASE_URL = "https://www.alphavantage.co/query";

let stockList = [];
let stockCharts = {}; // Store references to multiple stock charts

// 🔹 Load NSE & BSE stock symbols from a JSON file
async function loadStockSymbols() {
    try {
        const response = await fetch("indian_stocks.json"); // Path to your JSON file
        stockList = await response.json();
    } catch (error) {
        console.error("Error loading stock symbols:", error);
    }
}

loadStockSymbols();

// 🔹 Search function for NSE & BSE stocks
function searchStocks() {
    let query = document.getElementById("stock-symbol").value.trim().toUpperCase();
    let suggestionsDiv = document.getElementById("suggestions");

    if (query.length < 2) {
        suggestionsDiv.innerHTML = "";
        return;
    }

    let results = stockList
        .filter(stock => stock.name.toUpperCase().includes(query) || stock.symbol.toUpperCase().includes(query))
        .slice(0, 10);

    if (results.length === 0) {
        suggestionsDiv.innerHTML = "<div>No results found</div>";
        return;
    }

    let suggestions = results
        .map(stock => `<div onclick="selectStock('${stock.symbol}', '${stock.name}')">${stock.name} (${stock.symbol})</div>`)
        .join("");

    suggestionsDiv.innerHTML = suggestions;
}

// 🔹 Function to select a stock from suggestions
function selectStock(symbol, name) {
    document.getElementById("stock-symbol").value = `${name} (${symbol})`;
    document.getElementById("suggestions").innerHTML = "";
    fetchStockData(symbol);
}

// 🔹 Fetch real-time stock data from Alpha Vantage (with proper NSE/BSE format)
async function fetchStockData(symbol) {
    let nseSymbol = `${symbol}.NSE`;
    let bseSymbol = `${symbol}.BSE`;

    try {
        // Try NSE first
        let response = await fetch(`${API_BASE_URL}?function=GLOBAL_QUOTE&symbol=${nseSymbol}&apikey=${API_KEY}`);
        let stockData = await response.json();

        // If NSE data is empty, try BSE
        if (!stockData["Global Quote"] || Object.keys(stockData["Global Quote"]).length === 0) {
            response = await fetch(`${API_BASE_URL}?function=GLOBAL_QUOTE&symbol=${bseSymbol}&apikey=${API_KEY}`);
            stockData = await response.json();
        }

        if (!stockData["Global Quote"] || Object.keys(stockData["Global Quote"]).length === 0) {
            alert("Stock data not available. Try again later.");
            return;
        }

        let stock = stockData["Global Quote"];

        document.getElementById("stock-details").style.display = "block";
        document.getElementById("stock-name").innerText = `Stock: ${symbol}`;
        document.getElementById("stock-price").innerText = `Price: ₹${stock["05. price"]}`;
        document.getElementById("stock-change").innerText = `Change: ${stock["09. change"]} (${stock["10. change percent"]})`;
        document.getElementById("stock-volume").innerText = `Volume: ${stock["06. volume"]}`;

        fetchHistoricalData(nseSymbol);
    } catch (error) {
        console.error("Error fetching stock details:", error);
        alert("Failed to fetch stock details. Please try again.");
    }
}

// 🔹 Fetch historical stock data and update chart
async function fetchHistoricalData(symbol) {
    try {
        const response = await fetch(`${API_BASE_URL}?function=TIME_SERIES_DAILY&symbol=${symbol}&apikey=${API_KEY}`);
        const data = await response.json();

        if (!data["Time Series (Daily)"]) {
            console.error("No historical data available");
            return;
        }

        let historicalPrices = Object.entries(data["Time Series (Daily)"])
            .map(([date, prices]) => ({ date, close: parseFloat(prices["4. close"]) }))
            .slice(0, 30); // Get last 30 days

        updateChart(historicalPrices);
    } catch (error) {
        console.error("Error fetching historical stock data:", error);
    }
}

// Function to create and add a new chart for a stock
function createStockChart(symbol, priceData) {
    const chartContainer = document.getElementById("charts-container");

    // Check if a chart for this stock already exists
    if (stockCharts[symbol]) {
        stockCharts[symbol].destroy(); // Remove the previous chart instance
    }

    // Create a new canvas for the stock chart
    const canvas = document.createElement("canvas");
    canvas.id = `chart-${symbol}`;
    chartContainer.appendChild(canvas);

    const ctx = canvas.getContext("2d");

    // Create a new Chart instance
    stockCharts[symbol] = new Chart(ctx, {
        type: "line",
        data: {
            labels: priceData.map(data => data.date),
            datasets: [{
                label: `${symbol} Stock Price`,
                data: priceData.map(data => data.close),
                borderColor: "blue",
                borderWidth: 2,
                fill: false,
                pointRadius: 2,
                tension: 0.2
            }]
        },
        options: {
            responsive: true,
            plugins: {
                legend: { display: true }
            },
            scales: {
                x: { display: true, title: { display: true, text: "Date" } },
                y: { display: true, title: { display: true, text: "Price (₹)" } }
            }
        }
    });
}
